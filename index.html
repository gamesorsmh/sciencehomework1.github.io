// ── Password unlock elements ── const pwInput = document.getElementById('pw'); const enterBtn = document.getElementById('enterBtn'); const toggleBtn = document.getElementById('toggleBtn'); const lockSvg = document.getElementById('lockSvg'); const shackle = document.getElementById('shackle'); const lockBody = document.getElementById('lockBody'); const keyhole = document.getElementById('keyhole'); const subtitle = document.getElementById('subtitle'); const dots = document.querySelectorAll('.dot'); const errMsg = document.getElementById('errMsg'); const dashboard = document.getElementById('dashboard'); // ── Update dot fill based on input length ── pwInput.addEventListener('input', () => { const len = pwInput.value.length; dots.forEach((dot, i) => { dot.classList.toggle('filled', i < len); }); }); // ── Toggle password visibility ── toggleBtn.addEventListener('click', () => { if (pwInput.type === 'password') { pwInput.type = 'text'; } else { pwInput.type = 'password'; } }); // ── Unlock logic ── function unlock() { if (pwInput.value === CORRECT_PASSWORD) { // Animate lock open lockSvg.classList.remove('idle'); lockSvg.classList.add('unlocking'); shackle.classList.add('open'); lockBody.classList.add('open'); keyhole.classList.add('open'); subtitle.classList.add('open'); // Fill dots green dots.forEach(dot => dot.classList.add('success')); // Slide up dashboard after short delay setTimeout(() => { document.getElementById('card').classList.add('hide'); dashboard.classList.add('visible'); }, 800); } else { // Shake input and show error pwInput.classList.add('error'); errMsg.classList.add('show'); setTimeout(() => { pwInput.classList.remove('error'); errMsg.classList.remove('show'); }, 800); } } enterBtn.addEventListener('click', unlock); pwInput.addEventListener('keydown', (e) => { if (e.key === 'Enter') unlock(); }); // ── Game search filter ── const searchBar = document.getElementById('searchBar'); const gamePanels = document.querySelectorAll('.game-panel'); searchBar.addEventListener('input', () => { const query = searchBar.value.toLowerCase(); gamePanels.forEach(panel => { const label = panel.querySelector('.panel-label').textContent.toLowerCase(); panel.classList.toggle('hidden', !label.includes(query)); }); });
